import streamlit as st
import pandas as pd
import numpy as np
from geopy.geocoders import Nominatim
import requests
import time
from ortools.constraint_solver import pywrapcp
from ortools.constraint_solver import routing_enums_pb2
import folium
from streamlit_folium import st_folium
from math import radians, sin, cos, sqrt, atan2

st.set_page_config(page_title="Route Optimizer", page_icon="🚚", layout="wide")

st.title("🚚 Advanced Customer Route Optimizer")
st.markdown("**50+ Stops • 30 min On-Site • Global Day Time Window • Google Maps Export**")

uploaded_file = st.file_uploader("Upload Excel file with customer addresses", type=["xlsx", "xls"])

if uploaded_file:
    df = pd.read_excel(uploaded_file)
    st.success(f"✅ Loaded {len(df)} customers")
    st.dataframe(df.head(), use_container_width=True)

    address_col = st.selectbox("📍 Address Column", df.columns)

    st.subheader("🕒 Global Day Settings")
    col1, col2 = st.columns(2)
    with col1:
        day_start = st.time_input("Day Start Time", value=pd.to_datetime("08:00").time())
    with col2:
        day_end = st.time_input("Day End Time", value=pd.to_datetime("17:00").time())

    depot_address = st.text_input("🏢 Depot / Starting Address (Optional)", 
                                  placeholder="123 Business Park, Jackson, MS 39201")

    SERVICE_TIME_MIN = 30  # minutes on site per customer

    if st.button("🚀 Optimize Route", type="primary", use_container_width=True):
        with st.spinner("Geocoding addresses..."):
            geolocator = Nominatim(user_agent="route_optimizer")
            locations = []

            # Depot
            if depot_address.strip():
                try:
                    loc = geolocator.geocode(depot_address, timeout=10)
                    if loc:
                        locations.append({
                            "id": -1,
                            "address": depot_address,
                            "lat": loc.latitude,
                            "lon": loc.longitude,
                            "is_depot": True
                        })
                except:
                    st.warning("Could not geocode depot.")

            # Customers
            for idx, row in df.iterrows():
                try:
                    loc = geolocator.geocode(str(row[address_col]), timeout=10)
                    if loc:
                        locations.append({
                            "id": idx,
                            "address": str(row[address_col]),
                            "lat": loc.latitude,
                            "lon": loc.longitude,
                            "is_depot": False
                        })
                except:
                    st.warning(f"Failed to geocode: {row[address_col]}")
                time.sleep(1.1)

        if len(locations) < 2:
            st.error("Need at least 2 locations.")
            st.stop()

        st.success(f"✅ Geocoded {len(locations)} locations")

        # Build Time Matrix (driving + service time)
        with st.spinner("Calculating driving times..."):
            n = len(locations)
            time_matrix = np.zeros((n, n), dtype=int)

            for i in range(n):
                for j in range(n):
                    if i == j:
                        time_matrix[i][j] = 0
                        continue
                    try:
                        url = f"http://router.project-osrm.org/route/v1/driving/{locations[i]['lon']},{locations[i]['lat']};{locations[j]['lon']},{locations[j]['lat']}?overview=false"
                        r = requests.get(url, timeout=8).json()
                        duration_min = r['routes'][0]['duration'] / 60
                        time_matrix[i][j] = int(duration_min) + (SERVICE_TIME_MIN if not locations[j].get("is_depot") else 0)
                    except:
                        dist_km = haversine_distance(locations[i], locations[j])
                        time_matrix[i][j] = int(dist_km * 2) + (SERVICE_TIME_MIN if not locations[j].get("is_depot") else 0)

        # Solve
        with st.spinner("Solving optimal route..."):
            solution = solve_route(locations, time_matrix, day_start, day_end)

            if solution:
                route_indices = solution["route"]
                total_min = solution["total_time"]

                route_data = []
                for i, idx in enumerate(route_indices, 1):
                    loc = locations[idx]
                    route_data.append({
                        "Stop": i,
                        "Address": loc["address"],
                        "Type": "Depot" if loc.get("is_depot") else "Customer"
                    })

                route_df = pd.DataFrame(route_data)

                st.subheader("✅ Optimal Route")
                st.dataframe(route_df, use_container_width=True)

                st.metric("Total Route Time", f"{total_min} minutes (~{total_min//60}h {total_min%60}m)")

                # Google Maps
                addresses = [locations[i]["address"] for i in route_indices]
                maps_url = create_google_maps_url(addresses)
                st.markdown(f"[🚗 Open in Google Maps]({maps_url})", unsafe_allow_html=True)

                # Map
                m = create_folium_map([locations[i] for i in route_indices])
                st_folium(m, width=800, height=600, use_container_width=True)

                # Download
                st.download_button("📥 Download Route CSV", route_df.to_csv(index=False), "optimized_route.csv", "text/csv")
            else:
                st.error("No feasible route found within the day time window.")

# ===================== Helpers =====================

def haversine_distance(loc1, loc2):
    R = 6371.0
    lat1 = radians(loc1["lat"])
    lon1 = radians(loc1["lon"])
    lat2 = radians(loc2["lat"])
    lon2 = radians(loc2["lon"])
    dlon = lon2 - lon1
    dlat = lat2 - lat1
    a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
    c = 2 * atan2(sqrt(a), sqrt(1-a))
    return R * c

def solve_route(locations, time_matrix, day_start, day_end):
    depot_idx = 0 if locations[0].get("is_depot") else 0
    n = len(locations)

    manager = pywrapcp.RoutingIndexManager(n, 1, depot_idx)
    routing = pywrapcp.RoutingModel(manager)

    def time_callback(from_idx, to_idx):
        return time_matrix[manager.IndexToNode(from_idx)][manager.IndexToNode(to_idx)]

    transit_index = routing.RegisterTransitCallback(time_callback)
    routing.SetArcCostEvaluatorOfAllVehicles(transit_index)

    # Time dimension
    routing.AddDimension(transit_index, 30, 1440, True, "Time")
    time_dimension = routing.GetDimensionOrDie("Time")

    # Global time window for all customers
    start_minutes = day_start.hour * 60 + day_start.minute
    end_minutes = day_end.hour * 60 + day_end.minute

    for i, loc in enumerate(locations):
        if not loc.get("is_depot"):
            index = manager.NodeToIndex(i)
            time_dimension.CumulVar(index).SetRange(start_minutes, end_minutes)

    # Solver config
    search_params = pywrapcp.DefaultRoutingSearchParameters()
    search_params.first_solution_strategy = routing_enums_pb2.FirstSolutionStrategy.PATH_CHEAPEST_ARC
    search_params.local_search_metaheuristic = routing_enums_pb2.LocalSearchMetaheuristic.GUIDED_LOCAL_SEARCH
    search_params.time_limit.FromSeconds(45)

    solution = routing.SolveWithParameters(search_params)

    if solution:
        index = routing.Start(depot_idx)
        route = []
        total_time = 0
        while not routing.IsEnd(index):
            node = manager.IndexToNode(index)
            route.append(node)
            prev = index
            index = solution.Value(routing.NextVar(index))
            total_time += time_matrix[manager.IndexToNode(prev)][manager.IndexToNode(index)]
        route.append(manager.IndexToNode(index))
        return {"route": route, "total_time": total_time}
    return None

def create_google_maps_url(addresses):
    if len(addresses) < 2:
        return "#"
    origin = addresses[0].replace(" ", "+")
    dest = addresses[-1].replace(" ", "+")
    waypoints = "|".join(addr.replace(" ", "+") for addr in addresses[1:-1])
    return f"https://www.google.com/maps/dir/?api=1&origin={origin}&destination={dest}&waypoints={waypoints}&travelmode=driving"

def create_folium_map(route_locs):
    m = folium.Map(location=[route_locs[0]["lat"], route_locs[0]["lon"]], zoom_start=11)
    for i, loc in enumerate(route_locs):
        color = "red" if i == 0 or i == len(route_locs)-1 else "blue"
        folium.Marker(
            [loc["lat"], loc["lon"]],
            popup=f"<b>Stop {i+1}</b><br>{loc['address']}",
            icon=folium.Icon(color=color)
        ).add_to(m)
    points = [[loc["lat"], loc["lon"]] for loc in route_locs]
    folium.PolyLine(points, weight=4, color="#FF4B4B").add_to(m)
    return m
